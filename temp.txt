 3. PhysicsMonomial guard against zero coefficients

 File: src/types/physics_monomial.jl
 Lines: 54-62 (inner constructor)

 Current: Outer constructors filter zeros, but inner constructor allows zeros.

 Fix: Add filtering in inner constructor:
 function PhysicsMonomial{A,T}(
     coeffs::Vector{Int},
     monos::Vector{Monomial{A,T}}
 ) where {A<:Union{FermionicAlgebra,BosonicAlgebra},T<:Integer}
     @assert length(coeffs) == length(monos) "coeffs and monos must have same length"

     # Filter zero coefficients
     nonzero_mask = coeffs .!= 0
     if !all(nonzero_mask)
         coeffs = coeffs[nonzero_mask]
         monos = monos[nonzero_mask]
     end

     # Handle empty (zero) result
     if isempty(coeffs)
         return new{A,T}([0], [Monomial{A}(T[])])
     end

     new{A,T}(coeffs, monos)
 end

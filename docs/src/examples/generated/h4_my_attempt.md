# [H4-My-Attempt: D-Only V2RDM Moment Basis for Periodic H₄ (No Solve)](@id h4-my-attempt)

This page is deliberately narrow. It constructs the periodic H₄ `Nk = 2`
V2RDM problem in NCTSSoS with the moment basis

```math
B_D = \{1\} \cup \{a_i a_j : i < j\}.
```

That is **identity plus the ${}^{2}\!D$ pair-annihilator basis only**.
No ${}^{2}\!Q$ basis. No ${}^{2}\!G$ basis. No solver call.

The second half then adds `${}^{1}\!D`, `${}^{2}\!Q`, and `${}^{2}\!G` by
elimination: they are affine functions of `${}^{2}\!D`, not new moment-basis
rows. That is the important V2RDM trick. Normal ordering alone does not do it.

## Build the H₄ fermionic variables

The asset stores spatial-orbital integrals for two crystal momenta. We expand
each spatial orbital into spin-orbitals ordered as
`(k=0, ↑)`, `(k=0, ↓)`, `(k=1, ↑)`, `(k=1, ↓)`.

````julia
function build_h4_spin_orbital_variables(; nk::Int, norb::Int)
    registry, ((c_up_k0, c_up_k0_dag),
               (c_dn_k0, c_dn_k0_dag),
               (c_up_k1, c_up_k1_dag),
               (c_dn_k1, c_dn_k1_dag)) = create_fermionic_variables([
        ("c_up_k0", 1:norb),
        ("c_dn_k0", 1:norb),
        ("c_up_k1", 1:norb),
        ("c_dn_k1", 1:norb),
    ])

    vars = ((c_up_k0, c_up_k0_dag), (c_dn_k0, c_dn_k0_dag),
            (c_up_k1, c_up_k1_dag), (c_dn_k1, c_dn_k1_dag))

    ann = Dict(
        (0, :up) => c_up_k0,
        (0, :dn) => c_dn_k0,
        (1, :up) => c_up_k1,
        (1, :dn) => c_dn_k1,
    )
    dag = Dict(
        (0, :up) => c_up_k0_dag,
        (0, :dn) => c_dn_k0_dag,
        (1, :up) => c_up_k1_dag,
        (1, :dn) => c_dn_k1_dag,
    )

    spin_orbitals = NamedTuple[]
    for family in 0:2nk-1
        k = family ÷ 2
        spin = iseven(family) ? :up : :dn
        for orb in 1:norb
            push!(spin_orbitals, (; a = ann[(k, spin)][orb],
                                   adag = dag[(k, spin)][orb],
                                   k,
                                   spin,
                                   orb,
                                   spatial = k * norb + orb))
        end
    end

    return registry, vars, spin_orbitals
end

asset = load_nk2_asset()
````

## Write the objective as a `${}^{2}\!D` functional

A direct electronic Hamiltonian contains one-body monomials
`a†_p a_r`. A D-only moment basis cannot generate those moments. The V2RDM
route folds the one-body part into the reduced two-body Hamiltonian
`${}^{2}\!K`, so the objective uses only quartic monomials
`a†_p a†_q a_s a_r`.

`build_k2` returns the full spatial-orbital reduced Hamiltonian tensor. The
`0.5` below is the unrestricted-pair convention used by the asset helper.

````julia
function build_d_only_hamiltonian(k2, spin_orbitals)
    seed = spin_orbitals[2].adag * spin_orbitals[1].adag *
           spin_orbitals[1].a * spin_orbitals[2].a
    ham = (0.0 + 0.0im) * seed

    for p in eachindex(spin_orbitals), q in eachindex(spin_orbitals),
        r in eachindex(spin_orbitals), s in eachindex(spin_orbitals)

        spin_orbitals[p].spin == spin_orbitals[r].spin || continue
        spin_orbitals[q].spin == spin_orbitals[s].spin || continue

        coeff = 0.5 * k2[spin_orbitals[p].spatial,
                         spin_orbitals[q].spatial,
                         spin_orbitals[r].spatial,
                         spin_orbitals[s].spatial]
        iszero(coeff) && continue

        ham += coeff * spin_orbitals[p].adag * spin_orbitals[q].adag *
               spin_orbitals[s].a * spin_orbitals[r].a
    end

    return 0.5 * (ham + adjoint(ham))
end

k2 = build_k2(asset.h1e, asset.eri;
    nk = nk,
    n_active_orb = norb,
    n_total_electrons = asset.total_active_electrons,
)

ham = build_d_only_hamiltonian(k2, spin_orbitals);
````

## Build `B_D = identity + ²D`

The `${}^{2}\!D` rows are pair annihilators `a_i a_j` with `i < j`.
For 32 spin-orbitals this gives `binomial(32, 2) = 496` pair rows. We keep
the identity as a separate scalar block so the PSD footprint is easy to read.

````julia
function build_d_only_basis(spin_orbitals; nk::Int)
    NM = typeof(monomials(spin_orbitals[1].a * spin_orbitals[2].a)[1])
    identity_mono = one(NM)

    D_blocks = [NM[] for _ in 1:nk]
    for i in 1:length(spin_orbitals)-1, j in i+1:length(spin_orbitals)
        K = mod(spin_orbitals[i].k + spin_orbitals[j].k, nk)
        push!(D_blocks[K + 1], monomials(spin_orbitals[i].a * spin_orbitals[j].a)[1])
    end
    foreach(block -> sort!(unique!(block)), D_blocks)

    D2_basis = NM[identity_mono; reduce(vcat, D_blocks)]
    unique!(D2_basis)

    block_bases = Vector{NM}[[identity_mono]]
    append!(block_bases, D_blocks)

    return (; NM, identity_mono, D_blocks, D2_basis, block_bases)
end

basis_data = build_d_only_basis(spin_orbitals; nk);
````

`D2_basis` is the custom `moment_basis` passed to `SolverConfig`.

## Add only scalar sector constraints

We fix `N↑ = 4`, `N↓ = 4`, and `tr(²D) = binomial(8, 2) = 28`. These are
scalar moment equalities. They do **not** add `${}^{2}\!Q` or `${}^{2}\!G`
PSD bases.

````julia
function particle_number_operators(vars; norb::Int)
    (c_up_k0, c_up_k0_dag), (c_dn_k0, c_dn_k0_dag),
    (c_up_k1, c_up_k1_dag), (c_dn_k1, c_dn_k1_dag) = vars

    n_up = 1.0 * sum(c_up_k0_dag[i] * c_up_k0[i] +
                     c_up_k1_dag[i] * c_up_k1[i] for i in 1:norb)
    n_dn = 1.0 * sum(c_dn_k0_dag[i] * c_dn_k0[i] +
                     c_dn_k1_dag[i] * c_dn_k1[i] for i in 1:norb)
    return n_up, n_dn
end

function d_trace_operator(spin_orbitals, template)
    d_trace = zero(template)
    for i in 1:length(spin_orbitals)-1, j in i+1:length(spin_orbitals)
        d_trace += spin_orbitals[j].adag * spin_orbitals[i].adag *
                   spin_orbitals[i].a * spin_orbitals[j].a
    end
    return d_trace
end

n_up, n_dn = particle_number_operators(vars; norb)
D_trace = d_trace_operator(spin_orbitals, ham)
D_trace_target = binomial(asset.total_active_electrons, 2)

pop = polyopt(ham, registry;
    moment_eq_constraints = [
        n_up - 4.0 * one(ham),
        n_dn - 4.0 * one(ham),
        D_trace - D_trace_target * one(ham),
    ],
);
````

`pop` is an NCTSSoS polynomial optimization problem with fixed particle sector.

## Add `¹D`, `²Q`, and `²G` without adding variables

The NCTSSoS `MomentProblem` above is useful for checking the D-only moment
basis. But if we want `${}^{2}\!Q` and `${}^{2}\!G` **without increasing the
variables**, expanding `moment_basis` is the wrong lever.

Instead, keep only `${}^{2}\!D` as the free Hermitian variable and define

```math
{}^{1}\!D_{ps} = \frac{1}{N-1}\sum_q {}^{2}\!D_{pq,sq},
```

then build `${}^{2}\!Q(D)` and `${}^{2}\!G(D)` from the usual fermionic
anticommutation identities.

````julia
function build_pair_partition(spin_orbitals; nk::Int)
    keys = collect(0:nk-1)
    basis = Dict(K => Tuple{Int, Int}[] for K in keys)
    index = Dict(K => Dict{Tuple{Int, Int}, Int}() for K in keys)
    sector_of = Dict{Tuple{Int, Int}, Int}()

    for p in 1:length(spin_orbitals)-1, q in p+1:length(spin_orbitals)
        K = mod(spin_orbitals[p].k + spin_orbitals[q].k, nk)
        pair = (p, q)
        push!(basis[K], pair)
        index[K][pair] = length(basis[K])
        sector_of[pair] = K
    end

    return (; keys, basis, index, sector_of, block_sizes = [length(basis[K]) for K in keys])
end

function build_particle_hole_partition(spin_orbitals; nk::Int)
    keys = collect(0:nk-1)
    basis = Dict(K => Tuple{Int, Int}[] for K in keys)
    index = Dict(K => Dict{Tuple{Int, Int}, Int}() for K in keys)

    for p in eachindex(spin_orbitals), q in eachindex(spin_orbitals)
        Δk = mod(spin_orbitals[p].k - spin_orbitals[q].k, nk)
        pair = (p, q)
        push!(basis[Δk], pair)
        index[Δk][pair] = length(basis[Δk])
    end

    return (; keys, basis, index, block_sizes = [length(basis[K]) for K in keys])
end

canonical_pair(p::Int, q::Int) =
    p == q ? (0, (0, 0)) : p < q ? (1, (p, q)) : (-1, (q, p))

function zero_matrix_like(n::Int, prototype)
    mat = Matrix{typeof(prototype)}(undef, n, n)
    for j in 1:n, i in 1:n
        mat[i, j] = zero(prototype)
    end
    return mat
end

function build_eliminated_pqg_model(spin_orbitals, k2;
                                    nk::Int,
                                    n_electrons::Int)
    pair = build_pair_partition(spin_orbitals; nk)
    ph = build_particle_hole_partition(spin_orbitals; nk)

    model = Model()

    D_blocks = Dict{Int, Any}()
    for K in pair.keys
        n = length(pair.basis[K])
        D_blocks[K] = @variable(model, [1:n, 1:n] in HermitianPSDCone(),
                                base_name = "D_K$(K)")
    end

    prototype = D_blocks[first(pair.keys)][1, 1]

    function D_entry(p::Int, q::Int, r::Int, s::Int)
        sign_left, left_pair = canonical_pair(p, q)
        sign_right, right_pair = canonical_pair(r, s)
        (sign_left == 0 || sign_right == 0) && return zero(prototype)

        K_left = pair.sector_of[left_pair]
        K_right = pair.sector_of[right_pair]
        K_left == K_right || return zero(prototype)

        i = pair.index[K_left][left_pair]
        j = pair.index[K_right][right_pair]
        return sign_left * sign_right * D_blocks[K_left][i, j]
    end

    M = length(spin_orbitals)
    one_rdm = Matrix{typeof(prototype)}(undef, M, M)
    for p in 1:M, s in 1:M
        one_rdm[p, s] = sum(D_entry(p, q, s, q) for q in 1:M) / (n_electrons - 1)
    end

    Q_blocks = Dict{Int, Any}()
    for K in pair.keys
        basis = pair.basis[K]
        n = length(basis)
        mat = zero_matrix_like(n, prototype)
        for j in 1:n, i in 1:j
            p, q = basis[i]
            s, t = basis[j]

            expr = zero(prototype)
            expr += (p == s && q == t) ? 1.0 + 0.0im : 0.0 + 0.0im
            expr -= (p == t && q == s) ? 1.0 + 0.0im : 0.0 + 0.0im
            q == t && (expr -= one_rdm[s, p])
            p == t && (expr += one_rdm[s, q])
            q == s && (expr += one_rdm[t, p])
            p == s && (expr -= one_rdm[t, q])
            expr += D_entry(s, t, p, q)

            mat[i, j] = expr
        end
        Q_blocks[K] = Hermitian(mat, :U)
        @constraint(model, Q_blocks[K] in HermitianPSDCone())
    end

    G_blocks = Dict{Int, Any}()
    for Δk in ph.keys
        basis = ph.basis[Δk]
        n = length(basis)
        mat = zero_matrix_like(n, prototype)
        for j in 1:n, i in 1:j
            p, q = basis[i]
            s, t = basis[j]

            expr = zero(prototype)
            q == t && (expr += one_rdm[p, s])
            expr -= D_entry(p, t, s, q)

            mat[i, j] = expr
        end
        G_blocks[Δk] = Hermitian(mat, :U)
        @constraint(model, G_blocks[Δk] in HermitianPSDCone())
    end

    up_modes = [i for (i, mode) in enumerate(spin_orbitals) if mode.spin === :up]
    dn_modes = [i for (i, mode) in enumerate(spin_orbitals) if mode.spin === :dn]
    @constraint(model, real(sum(one_rdm[p, p] for p in up_modes)) == n_electrons ÷ 2)
    @constraint(model, real(sum(one_rdm[p, p] for p in dn_modes)) == n_electrons ÷ 2)

    D_trace_target = binomial(n_electrons, 2)
    Q_trace_target = binomial(M - n_electrons, 2)
    G_trace_target = n_electrons * (M - n_electrons + 1)

    @constraint(model,
        real(sum(D_blocks[K][i, i] for K in pair.keys for i in eachindex(pair.basis[K]))) ==
        D_trace_target)
    @constraint(model,
        real(sum(Q_blocks[K][i, i] for K in pair.keys for i in eachindex(pair.basis[K]))) ==
        Q_trace_target)
    @constraint(model,
        real(sum(G_blocks[K][i, i] for K in ph.keys for i in eachindex(ph.basis[K]))) ==
        G_trace_target)

    objective = zero(prototype)
    for p in eachindex(spin_orbitals), q in eachindex(spin_orbitals),
        r in eachindex(spin_orbitals), s in eachindex(spin_orbitals)

        spin_orbitals[p].spin == spin_orbitals[r].spin || continue
        spin_orbitals[q].spin == spin_orbitals[s].spin || continue

        coeff = 0.5 * k2[spin_orbitals[p].spatial,
                         spin_orbitals[q].spatial,
                         spin_orbitals[r].spatial,
                         spin_orbitals[s].spatial]
        iszero(coeff) && continue
        objective += coeff * D_entry(p, q, r, s)
    end
    @objective(model, Min, real(objective))

    return (; model, pair, ph, one_rdm, D_blocks, Q_blocks, G_blocks)
end

eliminated = build_eliminated_pqg_model(
    spin_orbitals,
    k2;
    nk = nk,
    n_electrons = asset.total_active_electrons,
);
````

## What this attempt does not claim

This page does **not** solve H₄ and does **not** claim a numerical ground-state
bound. It constructs the D-only moment-basis object and a separate eliminated
PQG coordinate model, then stops before handing either object to a solver.

This page earns its keep by making the variable accounting explicit and
checkable.
